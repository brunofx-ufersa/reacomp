Construtores

Estrutura simples

struct Pessoa
    nome::String
    idade::Int
    email::String
end

Criando instâncias

Vamos criar duas instâncias da estrutura Pessoa.

pessoa1 = Pessoa("Alice", 28, "alice@example.com")
pessoa2 = Pessoa("Sinfronio", 35, "sinfronio@example.com")
pessoa3 = Pessoa("Carol", 25, "carol@example.com")

println("Pessoa 1: ", pessoa1)
println("Pessoa 2: ", pessoa2)
println("Pessoa 3: ", pessoa3)
Pessoa 1: Pessoa("Alice", 28, "alice@example.com")
Pessoa 2: Pessoa("Sinfronio", 35, "sinfronio@example.com")
Pessoa 3: Pessoa("Carol", 25, "carol@example.com")

Acessando campos

Para acessar um campo de uma estrutura, usamos a sintaxe de ponto. Isto é, nome da instância da estrutura seguido de um ponto e o nome do campo, tudo junto.

println("Nome de pessoa1: ", pessoa1.nome)
println("Idade de pessoa1: ", pessoa1.idade)
println("Email de pessoa1: ", pessoa1.email)
Nome de pessoa1: Alice
Idade de pessoa1: 28
Email de pessoa1: alice@example.com

Estrutura mutável

Se tentarmos modificar um campo de pessoa2, por exemplo, vamos obter o seguinte erro:

immutable struct of type Pessoa cannot be changed

Isso significa literalmente que “a estrutura imutável do tipo Pessoa não pode ser alterada”. Por padrão, structs são imutáveis. Para contornar isso, podemos declarar usando a palavra-chave mutable, como no próximo exemplo.

mutable struct ContaBancaria
    titular::String
    saldo::Float64
end

conta = ContaBancaria("Maria", 1000.00)
println("Conta inicial: ", conta)
println("Saldo: R\$ ", conta.saldo)

# Modificando campos de struct mutável
conta.saldo += 500
println("Após depósito de R\$ 500: R\$ ", conta.saldo)

conta.saldo -= 200
println("Após saque de R\$ 200: R\$ ", conta.saldo)
Conta inicial: ContaBancaria("Maria", 1000.0)
Saldo: R$ 1000.0
Após depósito de R$ 500: R$ 1500.0
Após saque de R$ 200: R$ 1300.0

Estrutura com métodos

struct Ponto
    x::Float64
    y::Float64
    
    # Constructor customizado
    Ponto() = new(0.0, 0.0)  # Origin

    # Construtor para valores específicos
    Ponto(x, y) = new(x, y)
end
p1 = Ponto()  # Usa constructor customizado
println("Ponto padrão: (" , p1.x, ", ", p1.y, ")")
Ponto padrão: (0.0, 0.0)
# Exibição customizada
function Base.show(io::IO, p::Ponto)
    println(io, "(", p.x, ", ", p.y, ")")
end

p2 = Ponto(3.0, 4.0)
println("Ponto customizado: ", p2)
Ponto customizado: (3.0, 4.0)

Adicionando métodos

struct Retangulo
    largura::Float64
    altura::Float64
end
# Método: calcular área
area(r::Retangulo) = r.largura * r.altura
# Método: calcular perímetro
perimetro(r::Retangulo) = 2 * (r.largura + r.altura)
ret = Retangulo(5.0, 3.0)
println("Retângulo: ", ret)
println("Área: ", area(ret))
println("Perímetro: ", perimetro(ret))
Retângulo: Retangulo(5.0, 3.0)
Área: 15.0
Perímetro: 16.0

Herança

Vamos simular herança com composição.

struct Endereco
    rua::String
    numero::Int
    cidade::String
end

struct Funcionario
    nome::String
    departamento::String
    endereco::Endereco  # Composição
end
endereco = Endereco("Rua A", 123, "Fortaleza")
func = Funcionario("Carlos", "TI", endereco)

println("Funcionário: ", func.nome)
println("Departamento: ", func.departamento)
println("Endereço: ", func.endereco.rua, ", ", func.endereco.numero, ", ", func.endereco.cidade)
Funcionário: Carlos
Departamento: TI
Endereço: Rua A, 123, Fortaleza

Arrays de estruturas

pessoas = [ pessoa1, pessoa2, pessoa3 ]
println("Lista de pessoas:")
for pessoa in pessoas
    println("\t- ", pessoa.nome, " (", pessoa.idade, " anos)")
end

    - Alice (28 anos)
    - Sinfronio (35 anos)
    - Carol (25 anos)

Dicionário com estruturas

pessoas_dic = Dict(
    "id1" => pessoa1,
    "id2" => pessoa2,
    "id3" => pessoa3
)
println("Lista de pessoas:")
for (id, pessoa) in pessoas_dic
    println("\t", id, ": ", pessoa.nome, " (", pessoa.idade, " anos)")
end
Lista de pessoas:
    id1: Alice (28 anos)
    id2: Sinfronio (35 anos)
    id3: Carol (25 anos)

Exemplo prático

Vamos desenvolver um sistema simples de uma biblioteca.

struct Livro
    titulo::String
    autor::String
    ano_publicacao::Int
    isbn::String
end

mutable struct Biblioteca
    nome::String
    livros::Vector{Livro}
    
    # Constructor padrão
    Biblioteca(nome::String) = new(nome, Livro[])
end

Métodos para biblioteca

function adicionar_livro(bib::Biblioteca, livro::Livro)
    push!(bib.livros, livro)
end

function listar_livros(bib::Biblioteca)
    println("Biblioteca: ", bib.nome)
    println("Total de livros: ", length(bib.livros))
    for livro in bib.livros
        print("\t- ", livro.titulo)
        print(" por ", livro.autor)
        println(" (", livro.ano_publicacao, ")")
    end
end

Usando a biblioteca

bib = Biblioteca("Biblioteca Central")

adicionar_livro(bib, Livro("1984", "George Orwell", 1949, "978-0451524935"))
adicionar_livro(bib, Livro("O Senhor dos Anéis", "J.R.R. Tolkien", 1954, "978-0544003415"))
adicionar_livro(bib, Livro("Fundação", "Isaac Asimov", 1951, "978-0553293357"))
adicionar_livro(bib, Livro("O Hobbit", "J.R.R. Tolkien", 1937, "978-0547928227"))

listar_livros(bib)
Biblioteca: Biblioteca Central
Total de livros: 4

    - O Senhor dos Anéis por J.R.R. Tolkien (1954)
    - Fundação por Isaac Asimov (1951)
