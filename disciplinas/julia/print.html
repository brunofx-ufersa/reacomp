EXEMPLO PRÁTICO: FRACTAL MANDELBROT

println("\n— FRACTAL DE MANDELBROT —")

function mandelbrot(c, max_iter=100) z = 0 for n in 1:max_iter if abs(z) > 2 return n end z = z^2 + c end return max_iter end

Gerar grid de números complexos

width, height = 400, 400 xmin, xmax = -2.5, 1.0 ymin, ymax = -1.25, 1.25

mandelbrot_set = zeros(height, width)

for row in 1:height y = ymin + (row - 1) * (ymax - ymin) / (height - 1) for col in 1:width x = xmin + (col - 1) * (xmax - xmin) / (width - 1) c = complex(x, y) mandelbrot_set[row, col] = mandelbrot(c) end end

p_mandelbrot = heatmap(mandelbrot_set, title="Conjunto de Mandelbrot", xlabel="Real", ylabel="Imaginário", legend=false)

println("Fractal de Mandelbrot renderizado")

EXEMPLO: REGRESSÃO VISUAL

println("\n=== EXEMPLO: VISUALIZANDO REGRESSÃO ===")

Dados com ruído

x_reg = 1:20 y_reg = 2 .* x_reg .+ 5 .+ randn(20) .* 5

Calcular reta de regressão

x_meio = mean(x_reg) y_meio = mean(y_reg) b_reg = sum((x_reg .- x_meio) .* (y_reg .- y_meio)) / sum((x_reg .- x_meio).^2) a_reg = y_meio - b_reg * x_meio

Plot

p_reg = scatter(x_reg, y_reg, title="Dados com Regressão Linear", xlabel="X", ylabel="Y", label="Dados", markersize=6, alpha=0.7)

x_linha = 0:25 y_linha = a_reg .+ b_reg .* x_linha plot!(p_reg, x_linha, y_linha, label="Ajuste: y = $(round(a_reg; digits=2)) + $(round(b_reg; digits=2))x", linewidth=2, color=:red)

println("Plot de regressão criado")
